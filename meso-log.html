# MESO-LOG Deploy Script # Duplo-clique com botao direito → "Executar com PowerShell" $TOKEN = Read-Host "Cole seu token GitHub (ghp_...)" $FILE = Join-Path $PSScriptRoot "meso-log.html" $REPO = "marcowanuthe/meso-log" $PATH = "meso-log.html" if (-not (Test-Path $FILE)) { Write-Host "ERRO: meso-log.html nao encontrado em $PSScriptRoot" -ForegroundColor Red pause; exit 1 } Write-Host "Lendo arquivo..." -ForegroundColor Cyan $bytes = [System.IO.File]::ReadAllBytes($FILE) $b64 = [Convert]::ToBase64String($bytes) $size = [math]::Round($bytes.Length/1024) Write-Host "Arquivo: $size KB" -ForegroundColor Cyan Write-Host "Buscando SHA atual no GitHub..." -ForegroundColor Cyan $headers = @{ Authorization = "token $TOKEN"; Accept = "application/vnd.github.v3+json" } try { $info = Invoke-RestMethod "https://api.github.com/repos/$REPO/contents/$PATH" -Headers $headers $sha = $info.sha Write-Host "SHA atual: $($sha.Substring(0,10))..." -ForegroundColor Cyan } catch { $sha = $null Write-Host "Arquivo nao existe ainda, criando novo." -ForegroundColor Yellow } Write-Host "Enviando para GitHub Pages..." -ForegroundColor Cyan $body = @{ message = "fix: deploy meso-log correto via script" content = $b64 branch = "main" } if ($sha) { $body.sha = $sha } try { $result = Invoke-RestMethod "https://api.github.com/repos/$REPO/contents/$PATH" ` -Method PUT -Headers $headers ` -Body ($body | ConvertTo-Json -Depth 5) ` -ContentType "application/json" Write-Host "" Write-Host "✅ DEPLOY CONCLUIDO!" -ForegroundColor Green Write-Host "Commit: $($result.commit.sha.Substring(0,10))" -ForegroundColor Green Write-Host "" Write-Host "Aguarde ~1 minuto e acesse:" -ForegroundColor White Write-Host "https://marcowanuthe.github.io/meso-log/meso-log.html" -ForegroundColor Yellow } catch { Write-Host "ERRO no deploy: $_" -ForegroundColor Red } pause
